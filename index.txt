<h1 id="title">Marcello, coffee with me? ☕</h1>

<div class="buttons">
    <button id="yes" onclick="yesClicked()">YES 😋</button>
    <button id="no">NO 🙈</button>
</div>

<div id="timer">You have 30 seconds 😏</div>

<script>
const noBtn = document.getElementById("no");
const yesBtn = document.getElementById("yes");
const title = document.getElementById("title");

let scale = 1;
let yesScale = 1;
let noHoverCount = 0;

noBtn.addEventListener("mouseover", () => {
    noHoverCount++;

    const maxX = window.innerWidth - noBtn.offsetWidth;
    const maxY = window.innerHeight - noBtn.offsetHeight;

    const x = Math.random() * maxX;
    const y = Math.random() * maxY;

    scale -= 0.1;
    if (scale < 0.4) scale = 0.4;

    noBtn.style.position = "absolute";
    noBtn.style.left = x + "px";
    noBtn.style.top = y + "px";
    noBtn.style.transform = "scale(" + scale + ")";

    yesScale += 0.2;
    yesBtn.style.transform = "scale(" + yesScale + ")";

    if (noHoverCount >= 4) {
        yesBtn.style.position = "absolute";
        yesBtn.style.left = x + "px";
        yesBtn.style.top = y + "px";
        yesBtn.style.zIndex = "10";
        noHoverCount = 0;
    }
});

function createConfetti() {
    for (let i = 0; i < 100; i++) {
        const confetti = document.createElement("div");
        confetti.classList.add("confetti");

        confetti.style.left = Math.random() * 100 + "vw";
        confetti.style.backgroundColor =
            "hsl(" + Math.random() * 360 + ", 100%, 60%)";
        confetti.style.animationDuration = (Math.random() * 2 + 2) + "s";

        document.body.appendChild(confetti);
        setTimeout(() => confetti.remove(), 4000);
    }
}

function yesClicked() {
    createConfetti();

    document.body.innerHTML = `
        <div class="fade">
            <h1 style="font-size:48px;">YAY!! ☕😜</h1>
            <p style="font-size:24px;">Check your email 😉</p>
        </div>
    `;

    setTimeout(() => {
        window.location.href =
        "mailto:Petronellameleni01@gmail.com?subject=Coffee%20%3A)&body=Yes%20%E2%98%95%EF%B8%8F%E2%98%BA%EF%B8%8F";
    }, 1500);
}

let timeLeft = 30;
const timer = document.getElementById("timer");

const countdown = setInterval(() => {
    timeLeft--;

    timer.textContent = "You have " + timeLeft + " seconds 😏";

    if (timeLeft === 20) title.textContent = "Okay… just hovering? 👀";
    if (timeLeft === 15) title.textContent = "You know you want to say yes 😌☕";
    if (timeLeft === 10) title.textContent = "Marcello… don’t overthink it 😏";
    if (timeLeft === 5) title.textContent = "Last chance… 👀☕";

    yesScale += 0.05;
    yesBtn.style.transform = "scale(" + yesScale + ")";

    if (timeLeft <= 0) {
        clearInterval(countdown);
        timer.textContent = "I'll take that as a yes 😌";

        setTimeout(() => {
            yesClicked();
        }, 1000);
    }
}, 1000);
</script>
